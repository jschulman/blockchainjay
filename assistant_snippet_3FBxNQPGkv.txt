export const pageQuery = graphql`
  query IndexQueryPod {
    site {
      siteMetadata {
        title
        description
      }
    }
    allMarkdownRemark(
      limit: 1000
      filter: { 
        frontmatter: { 
          template: { eq: "post" }
          draft: { ne: true }
          audio: { eq: true }
        } 
      }
      sort: { frontmatter: { date: DESC } }
    ) {
      edges {
        node {
          fields {
            slug
            categorySlug
          }
          frontmatter {
            title
            date
            category
            description
            audio
            template
            attachments {
              publicURL
            }
          }
        }
      }
    }
  }
`
